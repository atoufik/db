        id_operateur                IN VARCHAR2,

        login                     IN VARCHAR2
    ) AS
    BEGIN
        INSERT INTO system.dossier_contentieux (
            "ID",

            flag_douane,
            avancement,
            agent
        ) VALUES (
            dossiercontentieux_seq.NEXTVAL,
            TO_DATE(date_saisie, 'YYYY-MM-DD HH24:MI:SS'),

            avancement,

        id_dossier := dossiercontentieux_seq.currval;
